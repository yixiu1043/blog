    # yamljs转换yml为json

    # 安装

    $ yarn add yamljs

    # 使用

    import YAML from 'yamljs';
import data from 'data.yml';

const json = YAML.load(data);

    # 其他

    // parse YAML string
nativeObject = YAML.parse(yamlString);
 
// Generate YAML
yamlString = YAML.stringify(nativeObject, 4);
